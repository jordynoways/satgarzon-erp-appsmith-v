CREATE TABLE IF NOT EXISTS public.proveedores (
  id BIGINT PRIMARY KEY,
  nombre_fiscal VARCHAR(255),
  nombre_comercial VARCHAR(255),
  cif_nif VARCHAR(50),
  direccion TEXT,
  poblacion VARCHAR(255),
  provincia VARCHAR(255),
  codigo_postal VARCHAR(10),
  telefono VARCHAR(50),
  email VARCHAR(255),
  referencia VARCHAR(100),
  id_stel BIGINT
);

CREATE INDEX IF NOT EXISTS idx_proveedores_nombre ON public.proveedores(nombre_fiscal);
CREATE INDEX IF NOT EXISTS idx_proveedores_cif ON public.proveedores(cif_nif);
