INSERT INTO public.proveedores (id, nombre_fiscal, nombre_comercial, cif_nif)
VALUES (
  {{ Number(this.params.id) }},
  {{ this.params.nombre_fiscal || 'Sin Nombre' }},
  {{ this.params.nombre_comercial || null }},
  {{ this.params.cif_nif || null }}
)
ON CONFLICT (id) 
DO UPDATE SET
  nombre_fiscal = EXCLUDED.nombre_fiscal,
  nombre_comercial = EXCLUDED.nombre_comercial,
  cif_nif = EXCLUDED.cif_nif;
